# Пример файла для массового удаления пользователей
# Формат: один email или номер телефона на строку
# Строки начинающиеся с # игнорируются (это комментарии)

# Удалить тестовых пользователей
test1@example.com
test2@example.com
+992901234567

# Удалить старых пользователей
olduser@mail.com
+992907654321

# Можно добавлять комментарии
user3@example.com  # Этот пользователь тоже будет удален
